3   import lombok.AllArgsConstructor;
4   import lombok.Data;
5   import lombok.NoArgsConstructor;
6   
7   import java.util.List;

9   /**
10   * Class representing the players position.
11   */
12  @NoArgsConstructor
13  @AllArgsConstructor
14  @Data
15  public class Position {
16      /**
17       * Indicates the first coordinate of the player.
18       */
19      int h;
20      /**
21       * Indicates the second coordinate of the player.
22       */
23      int w;
24  
25      /**
26       * Searches the {@code charMap} for the players coordinates, and
27       * sets the position to it.
28       *
29       * @param charMap a list of character lists, in which is to be searched for the {@code position}.
30       */
31      public void setPosition(List<List<Character>> charMap) {
32          for (int i = 0; i < charMap.size(); i++) {
33              if (charMap.get(i).contains('@')) {
34                  h = i;
35                  w = charMap.get(i).indexOf('@');
36              } else if (charMap.get(i).contains('+')) {
37                  h = i;
38                  w = charMap.get(i).indexOf('+');
39              }
40          }
41      }
42  }
