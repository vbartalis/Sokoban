54       */
55      public void newCharMap(){
56          logger.info("making new charmap");
57          charMap = new ArrayList<>();
58          for (int i = 0; i < stringMap.size(); i++) {
59              List<Character> characters = new ArrayList<>();
60              for (int j = 0; j < stringMap.get(i).length(); j++) {
61                  characters.add(stringMap.get(i).charAt(j));
62              }
63              charMap.add(characters);
64          }
65      }

105     /**
106      * Method for checking if a level is over.
107      * @return true if no finish blocks are free ({@literal "."}) and the player does not stay on a finish block ({@literal "+"}).
108      */
109     public boolean isLevelOver(){
110         boolean isOver = true;
111         for (List<Character> list: charMap) {
112             for (Character character: list) {
113                 if (character.equals('.') || character.equals('+')) isOver = false;
114             }
115         }
116         return isOver;
117     }
