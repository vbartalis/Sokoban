41      /**
42       * Method reads in a list of Levels from a json file.
43       * @return a list of Levels
44       */
45      public List<Level> levelFileReader(){
46          logger.info("reading level file");
47          Gson gson = new GsonBuilder().create();
48          Reader reader = new InputStreamReader(getClass().getResourceAsStream("/json/levels.json"));
49          Type type = new TypeToken<List<Level>>(){}.getType();
50          List<Level>levels = gson.fromJson(reader, type);
51          return levels;
52  
53      }

55      /**
56       * Method reads in a list of Saves from a json file, if no save file is found
57       * it creates a new list.
58       * @return list of Saves from a json file, if no save file is found
59       * it creates a new list and returns it.
60       */
61      public List<Save> saveFileReader(){
62          logger.info("trying to read save file");
63          List<Save> saves = new ArrayList<>();
64          try {
65              Gson gson = new GsonBuilder().create();
66              Reader reader = null;
67              reader = new FileReader("~/sokoban/save.json");
68              Type type = new TypeToken<List<Save>>(){}.getType();
69              saves = gson.fromJson(reader, type);
70              logger.info("save file successfully red");
71          } catch (FileNotFoundException e) {
72              logger.info("no save file found, creating new save");
73              Save nSaveve = new Save(levelNumber);
74              saves.add(newSave);
75          }
76          return saves;
77      }

79      /**
80       * Method try's to overwrite a list of saves into a file. If the file
81       * does not exist it creates it.
82       */
83      public void saveFileWriter(){
84          logger.info("trying to save");
85          try {
86              File file = new File("~/sokoban/save.json");
87              File parent = file.getParentFile();
88              if (!parent.exists() && !parent.mkdirs()) {
89                  throw new IllegalStateException("Couldn't create dir: " + parent);
90              }
91              file.createNewFile();
92              Gson gson = new GsonBuilder().setPrettyPrinting().create();
93              Writer writer = new FileWriter(file);
94  
95              Type type = new TypeToken<List<Save>>() {
96              }.getType();
97              String json = gson.toJson(saves, type);
98  
99              writer.write(json);
100             writer.close();
101             logger.info("file successfully saved");
102         } catch (IOException e) {
103             logger.error("couldn't write save file");
104             e.printStackTrace();
105         }
106 
107     }

109     /**
110      * Method for getting the {@code levelNumber} of the next element of the {@code levels} list, if there is one.

119     /**
120      * Method for getting the {@code levelNumber} of the previous element of the {@code levels} list, if there is one.

129     /**
130      * Method for setting up a new level and deleting the used {@code charMap} of the previous level.
131      * @param newLevelNumberString the {@code levelNumber} of the new level.
132      */
133     public void setNewLevel(String newLevelNumberString){
134         logger.info("setting up next level");
135         int newLevelNumberInt = Integer.parseInt(newLevelNumberString);
136         setLevelNumber(newLevelNumberInt);
137         levels.get(levelNumber).deleteCharMap();
138         levels.get(levelNumber).setStep(0);
139         levels.get(newLevelNumberInt).newCharMap();
140     }

142     /**
143      * Method for getting the best scale with which can be used to draw the level to the pane.
144      * @param pane the pane which is used to the drawing.
145      * @return the integer value that represents the best scale with which can be used
146      * to draw the level to the pane.
147      */
148     public int getScale(Pane pane){
149         logger.info("getting scale of game pane");
150         int scale;
151         double dispHeight = pane.getHeight();
152         double dispWidth = pane.getWidth();
153 
154         if (dispHeight == 0){
155             dispHeight = pane.getPrefHeight();
156         }
157         if (dispWidth == 0){
158             dispWidth = pane.getPrefWidth();
159         }
160 
161         int levelHeight = getLevel().getHeight();
162         int levelWidth = getLevel().getWidth();
163 
164         if (dispHeight/levelHeight < dispWidth/levelWidth){
165             scale = (int) (dispHeight / levelHeight);
166         }
167         else {
168             scale = (int) (dispWidth / levelWidth);
169         }
170         return scale;
171 
172     }

182     /**
183      * Returns the Save corresponding to the {@code levelNumber} index,
184      * if there is no such then it creates and returns a new empty Save.
185      * @return the Save corresponding to the {@code levelNumber} index,
186      * if there is no such then it creates and returns a new empty Save.

188     public Save getSave(){
189         logger.info("trying to get save of level {}", levelNumber);
190         if (!saves.isEmpty()) {
191             for (Save save : saves) {
192                 if (save.getLevelNumber() == levelNumber) {
193 
194                     return save;
195                 }
196             }
197         }
198         logger.info("no save found, making a new one");
199         Save nSaveve = new Save(levelNumber);
200         saves.add(newSave);
201         return newSave;
202     }
